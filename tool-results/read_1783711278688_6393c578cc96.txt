    72→    72→// ─── Component ─────────────────────────────────────────────────────
    73→    73→
    74→    74→export function ResumeStudioModule() {
    75→    75→  const { t } = useApp()
    76→    76→  const { toast } = useToast()
    77→    77→  const [rawText, setRawText] = useState('')
    78→    78→  const [jobDescription, setJobDescription] = useState('')
    79→    79→  const [showJD, setShowJD] = useState(false)
    80→    80→  const [result, setResult] = useState<StudioResult | null>(null)
    81→    81→  const [stage, setStage] = useState<PipelineStage>('idle')
    82→    82→  const [activeTab, setActiveTab] = useState('preview')
    83→    83→  const [editingSection, setEditingSection] = useState<string | null>(null)
    84→    84→  const [sectionBusy, setSectionBusy] = useState<string | null>(null)
    85→    85→
    86→    86→  const generate = async () => {
    87→    87→    if (!rawText.trim()) {
    88→    88→      toast({ title: 'Text required', description: 'Paste your raw text to generate a resume.', variant: 'destructive' })
    89→    89→      return
    90→    90→    }
    91→    91→    setResult(null)
    92→    92→    setStage('parsing')
    93→    93→
    94→    94→    // Animate through stages — slower to match actual AI latency
    95→    95→    setTimeout(() => setStage('enriching'), 5000)
    96→    96→    setTimeout(() => setStage('optimizing'), 15000)
    97→    97→    setTimeout(() => setStage('scoring'), 30000)
    98→    98→
    99→    99→    try {
   100→   100→      const res = await api<StudioResult>('/api/desktop/generate-resume-v2', {
   101→   101→        method: 'POST',
   102→   102→        body: { rawText, jobDescription: jobDescription || undefined, runQualityCheck: false },
   103→   103→      })
   104→   104→      setResult(res)
   105→   105→      setStage('done')
   106→   106→      setActiveTab('preview')
   107→   107→      const overall = res.evaluation?.overall ?? res.score?.overall ?? 0
   108→   108→      toast({ title: 'Resume generated! 🎉', description: `Overall score: ${overall}/100 — ${res.aiCalls ?? 1} AI call, ${res.latencyMs ?? 0}ms` })
   109→   109→    } catch (e) {
   110→   110→      toast({ title: 'Generation failed', description: (e as Error).message, variant: 'destructive' })
   111→   111→      setStage('idle')
   112→   112→    }
   113→   113→  }
   114→   114→
   115→   115→  const rewriteSection = async (section: string, content: any) => {
   116→   116→    setSectionBusy(section)
   117→   117→    try {
   118→   118→      const { result: rewritten } = await api<{ result: any }>('/api/desktop/rewrite-section', {
   119→   119→        method: 'POST', body: { section, content, jobDescription: jobDescription || undefined },
   120→   120→      })
   121→   121→      if (result) {
   122→   122→        const updated = { ...result, resume: { ...result.resume, [section]: rewritten } }
   123→   123→        setResult(updated)
   124→   124→        toast({ title: 'Section rewritten', description: `${section} updated with AI.` })
   125→   125→      }
   126→   126→    } catch (e) {
   127→   127→      toast({ title: 'Rewrite failed', description: (e as Error).message, variant: 'destructive' })
   128→   128→    }
   129→   129→    setSectionBusy(null)
   130→   130→  }
   131→   131→
   132→   132→  const translateResume = async (targetLang: 'ar' | 'en') => {
   133→   133→    if (!result) return
   134→   134→    setSectionBusy('translate')
   135→   135→    try {
   136→   136→      const { resume: translated } = await api<{ resume: any }>('/api/desktop/translate-resume', {
   137→   137→        method: 'POST', body: { resume: result.resume, targetLang },
   138→   138→      })
   139→   139→      setResult({ ...result, resume: translated })
   140→   140→      toast({ title: 'Resume translated', description: `Translated to ${targetLang === 'ar' ? 'Arabic' : 'English'}.` })
   141→   141→    } catch (e) {
   142→   142→      toast({ title: 'Translation failed', description: (e as Error).message, variant: 'destructive' })
   143→   143→    }
   144→   144→    setSectionBusy(null)
   145→   145→  }
   146→   146→
   147→   147→  const copyToClipboard = (text: string) => {
   148→   148→    navigator.clipboard.writeText(text)
   149→   149→    toast({ title: 'Copied', description: 'Content copied to clipboard.' })
   150→   150→  }
   151→   151→
   152→   152→  const exportJSON = () => {
   153→   153→    if (!result) return
   154→   154→    const blob = new Blob([JSON.stringify(result.resume, null, 2)], { type: 'application/json' })
   155→   155→    const url = URL.createObjectURL(blob)
   156→   156→    const a = document.createElement('a')
   157→   157→    a.href = url; a.download = 'resume.json'; a.click()
   158→   158→    URL.revokeObjectURL(url)
   159→   159→  }
   160→   160→
   161→   161→  const exportMarkdown = () => {
   162→   162→    if (!result) return
   163→   163→    const r = result.resume
   164→   164→    let md = `# ${r.contact?.name || 'Resume'}\n\n`
   165→   165→    if (r.contact?.email) md += `📧 ${r.contact.email} | `
   166→   166→    if (r.contact?.phone) md += `📱 ${r.contact.phone} | `
   167→   167→    if (r.contact?.location) md += `📍 ${r.contact.location}\n\n`
   168→   168→    if (r.objective) md += `## Objective\n${r.objective}\n\n`
   169→   169→    if (r.experience?.length) {
   170→   170→      md += `## Experience\n`
   171→   171→      r.experience.forEach((e: any) => {
   172→   172→        md += `### ${e.title || ''} — ${e.company || ''}\n${e.startDate || ''} - ${e.endDate || ''}\n`
   173→   173→        e.bullets?.forEach((b: string) => { md += `- ${b}\n` })
   174→   174→        md += '\n'
   175→   175→      })
   176→   176→    }
   177→   177→    if (r.education?.length) {
   178→   178→      md += `## Education\n`
   179→   179→      r.education.forEach((e: any) => { md += `### ${e.degree || ''} — ${e.school || ''}\n${e.startDate || ''} - ${e.endDate || ''}\n\n` })
   180→   180→    }
   181→   181→    if (r.skills?.technical?.length) md += `## Technical Skills\n${r.skills.technical.join(', ')}\n\n`
   182→   182→    if (r.skills?.soft?.length) md += `## Soft Skills\n${r.skills.soft.join(', ')}\n\n`
   183→   183→    if (r.skills?.languages?.length) md += `## Languages\n${r.skills.languages.map((l: any) => `${l.language}: ${l.level}`).join(', ')}\n\n`
   184→   184→    if (r.courses?.length) {
   185→   185→      md += `## Courses\n`
   186→   186→      r.courses.forEach((c: any) => { md += `- ${c.name} (${c.provider || ''}, ${c.hours || ''}, ${c.date || ''})\n` })
   187→   187→    }
   188→   188→
   189→   189→    const blob = new Blob([md], { type: 'text/markdown' })
   190→   190→    const url = URL.createObjectURL(blob)
   191→   191→    const a = document.createElement('a')
   192→   192→    a.href = url; a.download = 'resume.md'; a.click()
   193→   193→    URL.revokeObjectURL(url)
   194→   194→  }
   195→   195→
   196→   196→  const scoreColor = (score: number) => score >= 80 ? '#10b981' : score >= 60 ? '#f59e0b' : '#ef4444'
   197→   197→
   198→   198→  return (
   199→   199→    <div>
   200→   200→      <ModuleHeader title={t('studioTitle')} subtitle={t('studioSub')} icon={Sparkles}
   201→   201→        actions={result && (
   202→   202→          <div className="flex gap-2">
   203→   203→            <Button variant="outline" size="sm" className="rounded-full" onClick={exportJSON}><Download className="h-3.5 w-3.5" /> JSON</Button>
   204→   204→            <Button variant="outline" size="sm" className="rounded-full" onClick={exportMarkdown}><FileType className="h-3.5 w-3.5" /> MD</Button>
   205→   205→            <Button variant="outline" size="sm" className="rounded-full" onClick={() => window.print()}><FileText className="h-3.5 w-3.5" /> PDF</Button>
   206→   206→          </div>
   207→   207→        )}
   208→   208→      />
   209→   209→
   210→   210→      {/* Input section — shows when no result or during generation */}
   211→   211→      <AnimatePresence mode="wait">
   212→   212→        {stage === 'idle' && !result && (
   213→   213→          <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: -10 }}>
   214→   214→            <Card className="mb-4">
   215→   215→              <CardContent className="p-6">
   216→   216→                <div className="flex items-center gap-2 mb-3">
   217→   217→                  <Brain className="h-5 w-5 text-brand" />
   218→   218→                  <h3 className="font-semibold">Paste anything — we'll build a professional resume</h3>
   219→   219→                </div>
   220→   220→                <Textarea
   221→   221→                  value={rawText}
   222→   222→                  onChange={(e) => setRawText(e.target.value)}
   223→   223→                  rows={10}
   224→   224→                  placeholder="Paste your raw text here: WhatsApp export, OCR scan, LinkedIn copy, notes, old resume, mixed Arabic+English…"
   225→   225→                  className="mb-3"
   226→   226→                />
   227→   227→                <div className="flex flex-wrap items-center gap-2 mb-3">
   228→   228→                  <Badge variant="secondary" className="text-[10px]">📄 OCR text</Badge>
   229→   229→                  <Badge variant="secondary" className="text-[10px]">💬 WhatsApp export</Badge>
   230→   230→                  <Badge variant="secondary" className="text-[10px]">🌐 Bilingual AR/EN</Badge>
   231→   231→                  <Badge variant="secondary" className="text-[10px]">📋 LinkedIn copy</Badge>
   232→   232→                  <Badge variant="secondary" className="text-[10px]">📝 Random notes</Badge>
   233→   233→                  <Badge variant="secondary" className="text-[10px]">🔧 Broken formatting</Badge>
   234→   234→                </div>
   235→   235→
   236→   236→                {showJD && (
   237→   237→                  <div className="mb-3">
   238→   238→                    <Label className="text-xs text-muted-foreground">Job description (optional — enables ATS optimization)</Label>
   239→   239→                    <Textarea value={jobDescription} onChange={(e) => setJobDescription(e.target.value)} rows={4} className="mt-1" placeholder="Paste the job description for ATS keyword matching…" />
   240→   240→                  </div>
   241→   241→                )}
   242→   242→                {!showJD && (
   243→   243→                  <Button variant="ghost" size="sm" className="text-xs mb-3" onClick={() => setShowJD(true)}>
   244→   244→                    <Target className="h-3 w-3" /> Add job description for ATS optimization
   245→   245→                  </Button>
   246→   246→                )}
   247→   247→
   248→   248→                <Button onClick={generate} disabled={!rawText.trim()} className="w-full bg-brand text-brand-foreground hover:bg-brand/90 rounded-full h-12 text-base">
   249→   249→                  <Wand2 className="h-5 w-5" /> Generate Professional Resume
   250→   250→                </Button>
   251→   251→              </CardContent>
   252→   252→            </Card>
   253→   253→          </motion.div>
   254→   254→        )}
   255→   255→
   256→   256→        {/* Pipeline progress */}
   257→   257→        {stage !== 'idle' && stage !== 'done' && (
   258→   258→          <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }}>
   259→   259→            <Card>
   260→   260→              <CardContent className="p-8">
   261→   261→                <div className="space-y-4">
   262→   262→                  {STAGES.filter(s => s.id !== 'done').map((s, i) => {
   263→   263→                    const isActive = stage === s.id
   264→   264→                    const isPast = STAGES.findIndex(x => x.id === stage) > i
   265→   265→                    const Icon = s.icon
   266→   266→                    return (
   267→   267→                      <motion.div key={s.id} className="flex items-center gap-3"
   268→   268→                        animate={{ opacity: isActive ? 1 : isPast ? 0.5 : 0.3 }}>
   269→   269→                        <div className={`h-10 w-10 rounded-full flex items-center justify-center ${isActive ? 'bg-brand text-brand-foreground' : isPast ? 'bg-brand/20 text-brand' : 'bg-muted text-muted-foreground'}`}>
   270→   270→                          {isActive ? <Spinner /> : isPast ? <CheckCircle2 className="h-5 w-5" /> : <Icon className="h-5 w-5" />}
   271→   271→                        </div>
   272→   272→                        <div>
   273→   273→                          <div className="text-sm font-medium">{s.label}</div>
   274→   274→                          {isActive && <div className="text-xs text-brand">Processing…</div>}
   275→   275→                        </div>
   276→   276→                      </motion.div>
   277→   277→                    )
   278→   278→                  })}
   279→   279→                </div>
   280→   280→              </CardContent>
   281→   281→            </Card>
   282→   282→          </motion.div>
   283→   283→        )}
   284→   284→
   285→   285→        {/* Results */}
   286→   286→        {result && stage === 'done' && (
   287→   287→          <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}>
   288→   288→            <Tabs value={activeTab} onValueChange={setActiveTab}>
   289→   289→              <TabsList className="grid w-full max-w-lg grid-cols-4 mb-4">
   290→   290→                <TabsTrigger value="preview">Preview</TabsTrigger>
   291→   291→                <TabsTrigger value="score">Score</TabsTrigger>
   292→   292→                <TabsTrigger value="keywords">Keywords</TabsTrigger>
   293→   293→                <TabsTrigger value="missing">Missing Info</TabsTrigger>
   294→   294→              </TabsList>
   295→   295→
   296→   296→              {/* ─── Preview Tab ─── */}
   297→   297→              <TabsContent value="preview" className="space-y-4">
   298→   298→                <div className="flex flex-wrap gap-2 mb-2">
   299→   299→                  <Button variant="outline" size="sm" className="rounded-full" onClick={() => translateResume('ar')} disabled={sectionBusy === 'translate'}>
   300→   300→                    {sectionBusy === 'translate' ? <Spinner /> : <Languages className="h-3.5 w-3.5" />} Translate to Arabic
   301→   301→                  </Button>
   302→   302→                  <Button variant="outline" size="sm" className="rounded-full" onClick={() => translateResume('en')} disabled={sectionBusy === 'translate'}>
   303→   303→                    <Languages className="h-3.5 w-3.5" /> Translate to English
   304→   304→                  </Button>
   305→   305→                  <Button variant="outline" size="sm" className="rounded-full" onClick={() => { setResult(null); setStage('idle'); setRawText('') }}>
   306→   306→                    <RefreshCw className="h-3.5 w-3.5" /> New Resume
   307→   307→                  </Button>
   308→   308→                </div>
   309→   309→
   310→   310→                {/* Resume Preview Card */}
   311→   311→                <Card className="overflow-hidden">
   312→   312→                  <CardContent className="p-0">
   313→   313→                    <div className="bg-white text-neutral-900 p-6 sm:p-8 text-sm" style={{ fontFamily: 'var(--font-geist-sans)' }}>
   314→   314→                      {/* Header */}
   315→   315→                      <header className="border-b-2 pb-3 mb-4" style={{ borderColor: '#10b981' }}>
   316→   316→                        <h1 className="text-2xl font-bold">{result.resume.contact?.name || 'Your Name'}</h1>
   317→   317→                        <div className="flex flex-wrap gap-x-3 gap-y-0.5 text-xs text-neutral-600 mt-1">
   318→   318→                          {result.resume.contact?.email && <span>{result.resume.contact.email}</span>}
   319→   319→                          {result.resume.contact?.phone && <span>· {result.resume.contact.phone}</span>}
   320→   320→                          {result.resume.contact?.location && <span>· {result.resume.contact.location}</span>}
   321→   321→                          {result.resume.contact?.linkedin && <span>· {result.resume.contact.linkedin}</span>}
   322→   322→                        </div>
   323→   323→                      </header>
   324→   324→
   325→   325→                      {/* Objective */}
   326→   326→                      {result.resume.objective && (
   327→   327→                        <SectionCard title="Objective" onRewrite={() => rewriteSection('objective', result.resume.objective)} busy={sectionBusy === 'objective'} onCopy={() => copyToClipboard(result.resume.objective)}>
   328→   328→                          <p className="text-neutral-700">{result.resume.objective}</p>
   329→   329→                        </SectionCard>
   330→   330→                      )}
   331→   331→
   332→   332→                      {/* Experience */}
   333→   333→                      {result.resume.experience?.length > 0 && (
   334→   334→                        <SectionCard title="Experience" onRewrite={() => rewriteSection('experience', result.resume.experience)} busy={sectionBusy === 'experience'} onCopy={() => copyToClipboard(JSON.stringify(result.resume.experience))}>
   335→   335→                          <div className="space-y-3">
   336→   336→                            {result.resume.experience.map((exp: any, i: number) => (
   337→   337→                              <div key={i}>
   338→   338→                                <div className="flex items-baseline justify-between">
   339→   339→                                  <span className="font-semibold">{exp.title || 'Role'}</span>
   340→   340→                                  <span className="text-xs text-neutral-500">{exp.startDate} – {exp.endDate}</span>
   341→   341→                                </div>
   342→   342→                                {exp.company && <div className="text-neutral-600 text-xs">{exp.company}</div>}
   343→   343→                                <ul className="list-disc ps-4 mt-1 space-y-0.5 text-neutral-700">
   344→   344→                                  {exp.bullets?.map((b: string, j: number) => <li key={j}>{b}</li>)}
   345→   345→                                </ul>
   346→   346→                              </div>
   347→   347→                            ))}
   348→   348→                          </div>
   349→   349→                        </SectionCard>
   350→   350→                      )}
   351→   351→
   352→   352→                      {/* Education */}
   353→   353→                      {result.resume.education?.length > 0 && (
   354→   354→                        <SectionCard title="Education" onRewrite={() => rewriteSection('education', result.resume.education)} busy={sectionBusy === 'education'} onCopy={() => copyToClipboard(JSON.stringify(result.resume.education))}>
   355→   355→                          {result.resume.education.map((ed: any, i: number) => (
   356→   356→                            <div key={i} className="mb-1.5">
   357→   357→                              <span className="font-semibold">{ed.degree || 'Degree'}</span>
   358→   358→                              <span className="text-neutral-600"> · {ed.school || 'School'}</span>
   359→   359→                              <div className="text-xs text-neutral-500">{ed.startDate} – {ed.endDate}</div>
   360→   360→                            </div>
   361→   361→                          ))}
   362→   362→                        </SectionCard>
   363→   363→                      )}
   364→   364→
   365→   365→                      {/* Skills */}
   366→   366→                      {result.resume.skills && (
   367→   367→                        <SectionCard title="Skills" onRewrite={() => rewriteSection('skills', result.resume.skills)} busy={sectionBusy === 'skills'} onCopy={() => copyToClipboard(JSON.stringify(result.resume.skills))}>
   368→   368→                          {result.resume.skills.technical?.length > 0 && (
   369→   369→                            <div className="mb-2">
   370→   370→                              <span className="text-xs font-bold uppercase text-neutral-500">Technical: </span>
   371→   371→                              <span className="text-neutral-700">{result.resume.skills.technical.join(', ')}</span>
   372→   372→                            </div>
   373→   373→                          )}
   374→   374→                          {result.resume.skills.soft?.length > 0 && (
